Dealing with vector

// CPP program to create an empty vector
// and push values one by one.
#include <bits/stdc++.h>
using namespace std;

int main()
{
	// Create an empty vector
	vector<int> vect;

	vect.push_back(10);
	vect.push_back(20);
	vect.push_back(30);

	for (int x : vect)
		cout << x << " ";

	return 0;
}

Sort Vector  In Ascending Order


#include <bits/stdc++.h>
using namespace std;

int main()
{
	vector<int> v{ 1, 5, 8, 9, 6, 7, 3, 4, 2, 0 };

	sort(v.begin(), v.end());

	cout << "Sorted \n";
	for (auto x : v)
		cout << x << " ";

	return 0;
}

Sort Vector In Descending Order


#include <bits/stdc++.h>
using namespace std;

int main()
{
	vector<int> v{ 1, 5, 8, 9, 6, 7, 3, 4, 2, 0 };

	sort(v.begin(), v.end(), greater<int>());

	cout << "Sorted \n";
	for (auto x : v)
		cout << x << " ";

	return 0;
}
